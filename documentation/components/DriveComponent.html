import { Component, ElementRef, OnInit, ViewChild } from '@angular/core';
import { DomSanitizer } from '@angular/platform-browser';

@Component({
  selector: 'app-drive',
  templateUrl: './drive.component.html',
  styleUrls: ['./drive.component.css']
})
export class DriveComponent implements OnInit {
  @ViewChild('urlInput') urlInputRef: ElementRef;

  url = 'https://drive.google.com/embeddedfolderview?id=';
  urlGiven: boolean;
  expression: any = /[-a-zA-Z0-9@:%_\+.~#?&//=]{2,256}\.[a-z]{2,4}\b(\/[-a-zA-Z0-9@:%_\+.~#?&//=]*)?/gi;

  constructor(public sanitizer: DomSanitizer) { }

  ngOnInit() {
  }

  onGetURL() {
    const getUrl = this.urlInputRef.nativeElement.value;
    this.url = this.url + getUrl + '#list';
    this.urlGiven = true;
  }

}


<br><div *ngIf="!urlGiven" align="center">
  <h4 style="color: #3f51b5">Hov du har ikke tilknyttet en drive mappe endnu</h4>
<p>
  <mat-form-field class="full-width">
    <input type="text" id="url" matInput placeholder="URL til din drive-mappe." #urlInput>
  </mat-form-field>
  <button mat-button style="color: #3f51b5" (click)="onGetURL()">Tilknyt</button>
</p>
</div>

<iframe [src]="this.sanitizer.bypassSecurityTrustResourceUrl(this.url)" style="width:100%; height:600px; border:0;"></iframe>
